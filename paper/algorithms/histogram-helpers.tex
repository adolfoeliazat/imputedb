  \begin{algorithmic}
    \Require{$H$ is histogram, represented as a map from buckets to values.$val$ is a number (integer or floating point)}
    %\Ensure{}
    
     \Function{Total}{$H$}
     	\State $ct \gets 0$
	\For {$i \in dom(H)$}
        		\State $ct \gets ct + H[i]$
        	\EndFor
	\Return ct
     \EndFunction
    
        \Function{Add}{$H, val$}
        	\State $ct \gets \Call{Total}{H}$
    	\For{$i \in dom(H)$}
		\State $H[i] \gets H[i] +  val * (H[i] / ct)$ 
	\EndFor
    	\Return{$H$}
    \EndFunction
    

    \Function{ScaleBy}{$H, val$}
    	\For{$i \in dom(H)$}
		\State $H[i] \gets H[i] * val$ 
	\EndFor
    	\Return{$H$}
    \EndFunction
    
        \Function{ScaleTo}{$H, val$}
        	\State $ct \gets \Call{Total}{H}$
        	  \For{$i \in dom(H)$}
	  	\State $H[i] \gets H[i] * val / ct$ 
	\EndFor
    	\Return{$H$}
    \EndFunction
    
    
  \end{algorithmic}
  \caption{Helper functions used to the in-plan histogram updates}
  \label{algo:histogram-transformation-helpers}
%\end{algorithm}
